function saveData(event) { event.preventDefault(); // Prevent form submission const formData = new FormData(elements.rabbitForm); const rabbitData = {}; // Gather form data for (const [key, value] of formData.entries()) { rabbitData[key] = value; } // Check eligibility for mating based on date of birth const dateOfBirth = new Date(rabbitData.dateOfBirth); const currentDate = new Date(); const ageInMonths = (currentDate - dateOfBirth) / (1000 * 60 * 60 * 24 * 30); // Convert milliseconds to months if (ageInMonths >= 6) { rabbitData.eligibleForMating = true; } else { rabbitData.eligibleForMating = false; } // Ensure pregnancy data is saved correctly const isPregnant = formData.get('pregnantDoe'); if (isPregnant === 'yes') { rabbitData.pregnant = true; rabbitData.pregnancyStart = rabbitData.pregnancyStartDoe; rabbitData.dueDate = rabbitData.dueDateDoe; } else { rabbitData.pregnant = false; rabbitData.pregnancyStart = null; rabbitData.dueDate = null; } // Save the data in the list const cageNumber = elements.rabbitForm.dataset.cageNumber; rabbitDataList[cageNumber] = rabbitData; alert("Rabbit data saved successfully!"); // Notify user of successful save console.log(rabbitDataList); // Debugging: log saved data to console resetFields(); // Reset form fields after saving goHome(); // Return to home page }
